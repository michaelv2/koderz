def prime_length(string):
    n = len(string)
    if n < 2:
        return False
    if n == 2:
        return True
    if n % 2 == 0:
        return False
    limit = int(n ** 0.5)
    i = 3
    while i <= limit:
        if n % i == 0:
            return False
        i += 2
    return True
