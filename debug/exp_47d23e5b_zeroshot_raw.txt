from typing import List

def parse_nested_parens(paren_string: str) -> List[int]:
    groups = paren_string.split()
    depths: List[int] = []
    for g in groups:
        cur = 0
        max_depth = 0
        for ch in g:
            if ch == '(':
                cur += 1
                if cur > max_depth:
                    max_depth = cur
            elif ch == ')':
                cur -= 1
        depths.append(max_depth)
    return depths
