def histogram(test):
    if not isinstance(test, str):
        return {}
    tokens = test.split()
    if not tokens:
        return {}
    counts = {}
    for t in tokens:
        counts[t] = counts.get(t, 0) + 1

    max_count = max(counts.values()) if counts else 0
    if max_count == 0:
        return {}

    result = {}
    for key in counts:
        if counts[key] == max_count:
            result[key] = max_count
    return result
