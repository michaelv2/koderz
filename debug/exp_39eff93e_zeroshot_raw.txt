def is_prime(n):
    """Check if number n is prime."""
    if n <= 1 or (n % 2 == 0 and n > 2): 
        return False
    for i in range(3, int(n**0.5) + 1, 2):
        if n % i == 0:
            return False
    return True

def x_or_y(n, x, y):
    """Return the value of x if n is a prime number and the value of y otherwise."""
    return x if is_prime(n) else y
