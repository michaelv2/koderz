def circular_shift(x, shift):
    """Circular shift the digits of the integer x, shift the digits right by shift
    and return the result as a string.
    If shift > number of digits, return digits reversed.
    """
    s = str(x)
    sign = ''
    if s.startswith('-'):
        sign, s = '-', s[1:]
    n = len(s)
    if n == 0:
        return sign
    if shift > n:
        res = s[::-1]
    else:
        k = shift % n
        if k == 0:
            res = s
        else:
            res = s[-k:] + s[:-k]
    return sign + res
