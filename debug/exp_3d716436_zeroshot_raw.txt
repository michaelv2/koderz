import re

def valid_date(date):
    if not date:
        return False
    if not re.fullmatch(r'\d{2}-\d{2}-\d{4}', date):
        return False
    mm, dd, yyyy = date.split('-')
    try:
        m = int(mm); d = int(dd)
    except ValueError:
        return False
    if m < 1 or m > 12 or d < 1:
        return False
    if m == 2:
        return d <= 29
    if m in (4, 6, 9, 11):
        return d <= 30
    return d <= 31
