def check_if_last_char_is_a_letter(txt):
    if not isinstance(txt, str) or not txt:
        return False
    last_char = txt[-1]
    if not last_char.isalpha():
        return False
    last_space = txt.rfind(' ')
    if last_space == -1:
        return len(txt) == 1
    else:
        return len(txt[last_space + 1:]) == 1
