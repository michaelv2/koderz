def file_name_check(file_name):
    parts = file_name.split('.')
    
    # Check if there is exactly one dot
    if len(parts) != 2:
        return 'No'
    
    name_part, extension_part = parts
    
    # Check if the substring before the dot is not empty and starts with a letter
    if not name_part or not name_part[0].isalpha():
        return 'No'
    
    # Check if the substring after the dot is one of the allowed extensions
    if extension_part not in ['txt', 'exe', 'dll']:
        return 'No'
    
    # Count the number of digits in the name part
    digit_count = sum(c.isdigit() for c in name_part)
    
    # Check if there are more than three digits
    if digit_count > 3:
        return 'No'
    
    return 'Yes'
