def get_odd_collatz(n):
    """
    Given a positive integer n, return a sorted list that has the odd numbers in collatz sequence.
    """
    if n <= 0:
        raise ValueError("n must be a positive integer")
    curr = n
    odds = []
    while True:
        if curr % 2 == 1:
            odds.append(curr)
        if curr == 1:
            break
        if curr % 2 == 0:
            curr //= 2
        else:
            curr = 3 * curr + 1
    return sorted(set(odds))
