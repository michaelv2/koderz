from typing import List

def factorize(n: int) -> List[int]:
    """Return list of prime factors of given integer in ascending order, repeated per multiplicity."""
    factors: List[int] = []
    if n < 2:
        return factors

    while n % 2 == 0:
        factors.append(2)
        n //= 2

    p = 3
    while p * p <= n:
        while n % p == 0:
            factors.append(p)
            n //= p
        p += 2

    if n > 1:
        factors.append(n)

    return factors
