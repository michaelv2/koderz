def get_closest_vowel(word):
    vowels = "aeiouAEIOU"
    length = len(word)
    
    # Start from the second last character and move leftwards
    for i in range(length - 2, 0, -1):
        if word[i] in vowels:
            if word[i-1] not in vowels and word[i+1] not in vowels:
                return word[i]
    
    return ""
