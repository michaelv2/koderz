```python
from typing import List, Tuple
import math

def find_closest_elements(numbers: List[float]) -> Tuple[float, float]:
    nums = sorted(numbers)
    best_diff = math.inf
    best_pair = (nums[0], nums[1])
    for i in range(1, len(nums)):
        diff = nums[i] - nums[i - 1]
        if diff < best_diff:
            best_diff = diff
            best_pair = (nums[i - 1], nums[i])
    return best_pair
```
