def sort_numbers(numbers: str) -> str:
    """Sort space-delimited words representing digits zero through nine from smallest to largest."""
    value = {
        'zero': 0, 'one': 1, 'two': 2, 'three': 3, 'four': 4,
        'five': 5, 'six': 6, 'seven': 7, 'eight': 8, 'nine': 9
    }
    if not numbers:
        return ""
    tokens = numbers.split()
    tokens.sort(key=lambda w: value[w])
    return " ".join(tokens)
