def move_one_ball(arr):
    """Return True if arr can be rotated (right shifts) to become non-decreasing."""
    n = len(arr)
    if n <= 1:
        return True
    drops = 0
    for i in range(n):
        if arr[i] > arr[(i + 1) % n]:
            drops += 1
            if drops > 1:
                return False
    return True
