def is_simple_power(x, n):
    """Return True if there exists an integer k >= 0 such that n**k == x."""
    # handle simple/special cases
    if x == 1:
        return True
    if n == 1:
        return False
    if n == -1:
        return x in (1, -1)
    if n == 0:
        return x == 0
    curr = 1
    # multiply until magnitude exceeds target's magnitude
    while abs(curr) <= abs(x):
        curr *= n
        if curr == x:
            return True
    return False
