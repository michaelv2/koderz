def count_up_to(n):
    """Return the list of all prime numbers strictly less than n."""
    if n <= 2:
        return []
    primes = []
    for x in range(2, n):
        is_prime = True
        limit = int(x ** 0.5)
        for p in primes:
            if p > limit:
                break
            if x % p == 0:
                is_prime = False
                break
        if is_prime:
            primes.append(x)
    return primes
