def special_factorial(n):
    """The Brazilian factorial is defined as:
    brazilian_factorial(n) = n! * (n-1)! * (n-2)! * ... * 1!
    where n > 0
    """
    if n < 0:
        raise ValueError("n must be non-negative")
    result = 1
    fact = 1
    for i in range(1, n + 1):
        fact *= i
        result *= fact
    return result
