def pluck(arr):
    best_val = None
    best_idx = None
    for i, v in enumerate(arr):
        if v % 2 == 0:
            if best_val is None or v < best_val:
                best_val = v
                best_idx = i
    if best_val is None:
        return []
    return [best_val, best_idx]
