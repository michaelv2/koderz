```python
def cycpattern_check(a , b):
    if b == "":
        return True
    if len(b) > len(a):
        return False
    seen = set()
    n = len(b)
    for i in range(n):
        rot = b[i:] + b[:i]
        if rot in seen:
            continue
        seen.add(rot)
        if rot in a:
            return True
    return False
```
