def encode(message):
    vowels = "aeiou"
    shifted_vowels = {v: chr((ord(v) - 97 + 2) % 26 + 97) for v in vowels}
    shifted_vowels.update({v.upper(): shifted_vowels[v].upper() for v in vowels})
    
    def swap_case_and_shift_vowel(char):
        if char.isalpha():
            if char.lower() in shifted_vowels:
                return shifted_vowels[char]
            else:
                return char.swapcase()
        return char
    
    return ''.join(swap_case_and_shift_vowel(c) for c in message)
