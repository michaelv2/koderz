```python
def solve(s):
    if not any(char.isalpha() for char in s):
        return s[::-1]
    
    result = ""
    for char in s:
        if char.isalpha():
            if char.islower():
                result += char.upper()
            else:
                result += char.lower()
        else:
            result += char
            
    return result
```
