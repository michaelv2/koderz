def match_parens(lst):
    # Concatenate the two strings and count the number of open and close parentheses
    total = lst[0] + lst[1]
    open_count = total.count('(')
    close_count = total.count(')')
    
    # If the counts are equal, it means there's a way to make a good string
    if open_count == close_count:
        return 'Yes'
    else:
        return 'No'
