I will check each number for primality (ignoring numbers <= 1), track the largest prime found, and return the sum of its decimal digits (or 0 if no prime is present).

```python
import math

def skjkasdkd(lst):
    def is_prime(n):
        if n <= 1:
            return False
        if n <= 3:
            return True
        if n % 2 == 0 or n % 3 == 0:
            return False
        # check divisors of form 6k +/- 1 up to sqrt(n)
        limit = math.isqrt(n)
        i = 5
        while i <= limit:
            if n % i == 0 or n % (i + 2) == 0:
                return False
            i += 6
        return True

    max_prime = None
    for x in lst:
        if x is None:
            continue
        if x > 1 and is_prime(x):
            if max_prime is None or x > max_prime:
                max_prime = x

    if max_prime is None:
        return 0
    return sum(int(d) for d in str(abs(max_prime)))
```
