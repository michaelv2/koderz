def histogram(test):
    tokens = test.split()
    if not tokens:
        return {}
    counts = {}
    for t in tokens:
        counts[t] = counts.get(t, 0) + 1

    max_count = max(counts.values())
    if max_count == 0:
        return {}

    return {letter: cnt for letter, cnt in counts.items() if cnt == max_count}
