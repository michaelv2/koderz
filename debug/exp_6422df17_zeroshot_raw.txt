from typing import List

def separate_paren_groups(paren_string: str) -> List[str]:
    """Input is a string containing multiple groups of nested parentheses.
    Return a list of separate balanced top-level parenthesis groups, ignoring spaces."""
    res: List[str] = []
    curr: List[str] = []
    balance = 0
    for ch in paren_string:
        if ch == ' ':
            continue
        if ch not in '()':
            continue
        curr.append(ch)
        if ch == '(':
            balance += 1
        else:
            balance -= 1
        if balance == 0 and curr:
            res.append(''.join(curr))
            curr = []
    return res
