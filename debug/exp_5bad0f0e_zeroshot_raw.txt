def pluck(arr):
    even_values = [i for i in range(len(arr)) if arr[i] % 2 == 0]
    
    if not even_values:
        return []
        
    smallest_even_value = min([arr[i] for i in even_values])
    smallest_index = [i for i in even_values if arr[i] == smallest_even_value][0]
    
    return [smallest_even_value, smallest_index]
