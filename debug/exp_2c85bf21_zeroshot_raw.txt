def int_to_mini_roman(number):
    val = [
        1000, 900, 500, 400,
        100, 90, 50, 40,
        10, 9, 5, 4,
        1
        ]
    syms = [
        "m", "cm", "d", "cd",
        "c", "xc", "l", "xl",
        "x", "ix", "v", "iv",
        "i"
        ]
    roman_numeral = ''
    i = 0
    while number > 0:
        for _ in range(number // val[i]):
            roman_numeral += syms[i]
            number -= val[i]
        i += 1
    return roman_numeral
