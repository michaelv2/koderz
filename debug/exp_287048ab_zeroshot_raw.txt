def greatest_common_divisor(a: int, b: int) -> int:
    """ Return a greatest common divisor of two integers a and b """
    a, b = abs(a), abs(b)
    # If both are zero, gcd is conventionally 0
    if a == 0 and b == 0:
        return 0
    while b:
        a, b = b, a % b
    return a
