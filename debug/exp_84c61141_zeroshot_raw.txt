def iscube(a):
    if a >= 0:
        r = round(a ** (1/3))
        start = max(0, int(r) - 2)
        end = int(r) + 3
        for c in range(start, end):
            if c ** 3 == a:
                return True
        return False
    else:
        b = -a
        r = round(b ** (1/3))
        start = max(0, int(r) - 2)
        end = int(r) + 3
        for c in range(start, end):
            if -(c ** 3) == a:
                return True
        return False
