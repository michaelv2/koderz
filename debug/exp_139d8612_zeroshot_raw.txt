I will iterate through each parenthesis group (split by whitespace), track current depth and maximum depth per group, and return the list of maxima.

```python
from typing import List

def parse_nested_parens(paren_string: str) -> List[int]:
    """Return the maximum nesting depth for each group of parentheses separated by whitespace."""
    if not paren_string or paren_string.strip() == "":
        return []
    groups = paren_string.split()
    result: List[int] = []
    for grp in groups:
        depth = 0
        max_depth = 0
        for ch in grp:
            if ch == '(':
                depth += 1
                if depth > max_depth:
                    max_depth = depth
            elif ch == ')':
                depth -= 1
                if depth < 0:
                    # malformed, but continue (treat negative as zero)
                    depth = 0
            else:
                # ignore other characters
                continue
        result.append(max_depth)
    return result
```
