def even_odd_count(num):
    """Given an integer. return a tuple that has the number of even and odd digits respectively.

     Example:
        even_odd_count(-12) ==> (1, 1)
        even_odd_count(123) ==> (1, 2)
    """
    s = str(abs(int(num)))
    ev = sum(1 for ch in s if (ord(ch) - ord('0')) % 2 == 0)
    od = len(s) - ev
    return (ev, od)
