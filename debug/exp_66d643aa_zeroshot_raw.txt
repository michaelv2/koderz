def car_race_collision(n: int):
    """
    Calculate the number of collisions between two sets of n cars moving in opposite directions
    on an infinite straight road, where all cars have the same speed and continue after
    collisions (i.e., they pass through each other). Each pair consisting of one car from
    the left-to-right group and one from the right-to-left group will collide exactly once,
    so the total number of collisions is n * n.
    """
    return n * n
