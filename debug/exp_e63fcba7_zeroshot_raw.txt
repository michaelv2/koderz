def check_if_last_char_is_a_letter(txt):
    # Check if the string is empty or not
    if txt == "":
        return False
    
    # Split the text into words and get the last word
    words = txt.split()
    if len(words) == 0:
        return False
    last_word = words[-1]
    
    # Check if the last character of the last word is an alphabetical character
    return last_word[-1].isalpha() and (len(last_word) == 1 or not last_word[-2].isalpha())
