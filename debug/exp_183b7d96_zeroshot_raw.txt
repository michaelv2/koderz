def by_length(arr):
    # Filter the array to include only numbers between 1 and 9 inclusive
    filtered_arr = [x for x in arr if 1 <= x <= 9]
    
    # Sort the filtered array
    sorted_arr = sorted(filtered_arr)
    
    # Reverse the sorted array
    reversed_arr = sorted_arr[::-1]
    
    # Map numbers to their corresponding names
    number_names = {
        1: "One",
        2: "Two",
        3: "Three",
        4: "Four",
        5: "Five",
        6: "Six",
        7: "Seven",
        8: "Eight",
        9: "Nine"
    }
    
    # Convert numbers to their names
    result = [number_names[x] for x in reversed_arr]
    
    return result
